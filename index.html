# Rename YYYY.MM.DD to YYYY-MM-DD format
for file in *.txt; do
  newname=$(echo "$file" | sed -E 's/(\d{4})\.(\d{2})\.(\d{2})/\1-\2-\3/')
  mv "$file" "$newname"
done
